**Goal:** A per-article table of contents — sticky side-rail with scroll-spy on `xl+`, a `<details>` disclosure on narrow screens.

**Architecture:** A pure `extractToc` parses h2/h3 from the MDX (ids via `github-slugger`, matching `rehype-slug`). **Tech Stack:** Next.js 16, Tailwind v4, `github-slugger`, `rehype-slug` (existing), Vitest.

import GithubSlugger from "github-slugger";

export interface TocItem {
  depth: 2 | 3;
  text: string;
  id: string;
}

function stripInline(s: string): string {
  return s
    .replace(/`([^`]+)`/g, "$1")
    .replace(/\*\*([^*]+)\*\*/g, "$1")
    .replace(/\*([^*]+)\*/g, "$1")
    .replace(/_([^_]+)_/g, "$1")
    .replace(/\[([^\]]+)\]\([^)]+\)/g, "$1")
    .trim();
}

/** h2/h3 headings from MDX, with rehype-slug-compatible ids. Skips code fences. */
export function extractToc(content: string): TocItem[] {
  const slugger = new GithubSlugger();
  const items: TocItem[] = [];
  let inFence = false;

  for (const line of content.split("\n")) {
    if (/^\s*```/.test(line)) {
      inFence = !inFence;
      continue;
    }
    if (inFence) continue;

    const m = /^(#{2,3})\s+(.+?)\s*#*\s*$/.exec(line);
    if (!m) continue;

    const depth = m[1].length as 2 | 3;
    const text = stripInline(m[2]);
    items.push({ depth, text, id: slugger.slug(text) });
  }

  return items;
}
```

"use client";

import { useEffect, useState } from "react";
import type { TocItem } from "@/lib/toc";

export function TableOfContents({ items }: { items: TocItem[] }) {
  const [activeId, setActiveId] = useState("");

  useEffect(() => {
    const headings = items
      .map((i) => document.getElementById(i.id))
      .filter((el): el is HTMLElement => el !== null);
    if (headings.length === 0) return;

    const observer = new IntersectionObserver(
      (entries) => {
        const visible = entries
          .filter((e) => e.isIntersecting)
          .sort((a, b) => a.boundingClientRect.top - b.boundingClientRect.top);
        if (visible[0]) setActiveId(visible[0].target.id);
      },
      { rootMargin: "0px 0px -70% 0px", threshold: 0 },
    );

    headings.forEach((h) => observer.observe(h));
    return () => observer.disconnect();
  }, [items]);

  if (items.length === 0) return null;

  return (
    <nav aria-label="Table of contents" className="text-sm">
      <p className="mb-3 font-display text-xs font-semibold uppercase tracking-wider text-faint">
        On this page
      </p>
      <ul className="border-l border-border">
        {items.map((item) => (
          <li key={item.id}>
            <a
              href={`#${item.id}`}
              className={`-ml-px block border-l-2 py-1 transition-colors ${
                item.depth === 3 ? "pl-6" : "pl-3"
              } ${
                activeId === item.id
                  ? "border-terracotta text-terracotta"
                  : "border-transparent text-muted hover:text-ink"
              }`}
            >
              {item.text}
            </a>
          </li>
        ))}
      </ul>
    </nav>
  );
}
```

- Id parity rests on both sides using `github-slugger` over the same cleaned text; Task 4's build grep is the guard. `rehype-slug` (already in `Mdx`) slugs rendered heading text; `stripInline` reproduces that text from the markdown source for the common cases (code, emphasis, links).
- The article body contains a literal `# Hello World` h1 in addition to the frontmatter title; `extractToc` correctly skips h1s, so the ToC lists only the `##`/`###` sections.
- The `xl` grid keeps prose at ~48rem (max-w-5xl minus the 13rem rail and gap); below `xl` the article is plain `max-w-3xl` and the `<details>` provides the ToC.
- Scroll-spy sets state only inside the IntersectionObserver callback (async), so it does not trip `react-hooks/set-state-in-effect`.
